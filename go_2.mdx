```go subtitle="Go has a specific data type for complex numbers" title="Complex data type"
package main

import(
    "fmt"
    "reflect"
    "unsafe"
)

func main(){
    var a float32 = 3
    var b float32 = 5
    
    //Initialize-1
    c := complex(a, b)//saved as a+bi and as a complex64 type
    
    //Initialize-2
    var d complex64
    d = 4 + 5i
    var e complex128
    fmt.Println(c)
    fmt.Println(d)
    //Print Size
    fmt.Printf("c's size is %d bytes\n", unsafe.Sizeof(c))
    fmt.Printf("d's size is %d bytes\n", unsafe.Sizeof(d))
    fmt.Printf("e's size is %d bytes\n", unsafe.Sizeof(e))
    //Print type
    fmt.Printf("c's type is %s\n", reflect.TypeOf(c))
    fmt.Printf("d's type is %s\n", reflect.TypeOf(d))
    fmt.Printf("e's type is %s\n", reflect.TypeOf(e))
    
    //Operations on complex number
    fmt.Println(c+d, c-d, c*d, c/d)
}
```

```go title="Byte data type"
package main

import(
	"fmt"
	"reflect"
	"unsafe"
)

func main() {
	var r byte = 'r'
	var b = ":	 ;."//string
	s := "abc" //string
	//Print Size
	fmt.Printf("Size: %d\n", unsafe.Sizeof(r))
	fmt.Printf("Size: %d\n", unsafe.Sizeof(b))

	//Print Type
	fmt.Printf("Type: %s\n", reflect.TypeOf(r))
	fmt.Printf("Type: %s\n", reflect.TypeOf(b))

	//Print Character
	fmt.Printf("Character: %c\n", r)

	fmt.Println(byte(r))
	fmt.Println([]byte(b))//output as bytes
	fmt.Println([]byte(s))// output as bytes
}
```

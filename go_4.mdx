```go title="Functions with multiple Returns"

    fmt.Println(b)

    fmt.Println(c)
```

## Variadic Functions

---

<CodeSurfer>

```go title="Variadic Functions"
package main

import "fmt"

func sum(nums ...int){
    fmt.Print(nums, " ")
    total := 0
    for _, num := range nums {
        total += num
    }
    fmt.Println(total)
}

func main() {
    sum(1, 2)
    sum(1, 2, 3)
}
```

```diff 5[9:22] subtitle="The ellipsis(...) is used for defining Variadic Functions. This function can take an arbitrary number of arguments of int data type her"
```

## Recursive Functions

---

<CodeSurfer>

```go subtitle="A function calling itself is called a recursive function"
package main

import "fmt"

func fact(n int) int {
    if n == 0 {
        return 1
    }
    return n * fact(n-1)
}

func main() {
    fmt.Println(fact(7))
}
```
```diff 5 subtitle="We're defining the fact function here with an int arg and int return type"
```
